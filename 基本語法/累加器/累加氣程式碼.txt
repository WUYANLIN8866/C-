#include <iostream>

/* run this program using the console pauser or add your own getch, system("pause") or input loop */

int main(int argc, char** argv) {
	int i, sum=0;
	int y;
	printf("請輸入累加數字 = ");
	scanf("%d",&y);
	for(i=1;i<=y;i++)
	{
		printf("%d",i);
		if(i<y)
		printf("+");
		sum+=i;	
	}
	printf(" = %d",sum);
system("pause");
	return 0;
}
